import { useEffect, useState } from "react";
import axios from "axios";
import { useNavigate } from "react-router-dom";

const HomePage = () => {
  const [books, setBooks] = useState([]);
  const [visibleBooks, setVisibleBooks] = useState(12);
  const navigate = useNavigate(); // Hook for navigation

  useEffect(() => {
    axios
      .get("/data/books.json")
      .then((res) => {
        if (Array.isArray(res.data)) {
          setBooks(res.data);
        } else {
          console.error("Invalid books data:", res.data);
        }
      })
      .catch((err) => console.error("Error fetching books:", err));
  }, []);

  const loadMoreBooks = () => {
    setVisibleBooks((prev) => prev + 12);
  };

  return (
    <div className="home">
      <h2 className="explore-header">ðŸ“š Explore Books</h2>

      <div className="books-grid">
        {books.slice(0, visibleBooks).map((book) => (
          <div 
            key={book.id} 
            className="book-card" 
            onClick={() => navigate(`/book/${book.id}`)}
          >
            <img src={book.coverImage} alt={book.title} />
            <h3>{book.title}</h3>
            <p>{book.price}</p>
          </div>
        ))}
      </div>

      {visibleBooks < books.length && (
        <div className="load-more-container">
          <button className="load-more" onClick={loadMoreBooks}>
            Load More
          </button>
        </div>
      )}
    </div>
  );
};

export default HomePage;
